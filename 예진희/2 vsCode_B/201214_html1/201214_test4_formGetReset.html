    <h1>Get방식</h1>
    용도: 검색에서 주로 쓰임<br>
    <form method="get">
        <input type="text" name="search">
        <input type="submit">
    </form>
 
    <h1>Post방식</h1>
    용도: 로그인시 쓰임<br>
    <form method="post">
        <input type="text" name="search">
        <input type="submit">
    </form>
    
 
    <h1>Reset 실습</h1>
    <form>
        <input type="text" name="search">
        <input type="submit">
        <input type="reset" value="지운다">
        <!-- value값 넣으면 버튼 txt바뀜 -->
    </form>
